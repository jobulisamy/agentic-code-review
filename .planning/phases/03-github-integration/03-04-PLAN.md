    ```python
    import json
    import logging
    from sqlalchemy import select
    from app.db.engine import AsyncSessionLocal
    from app.models.repo import Repo
    from app.models.review import Review
    from app.pipeline.orchestrator import run_review, ReviewPipelineError
    from app.services.github import (
        get_installation_token, fetch_pr_diff, build_diff_comment_positions,
        finding_to_comment, format_summary_comment, submit_review, post_failure_comment,
    )
    from unidiff import PatchSet

    async def run_webhook_review(payload: dict, settings) -> None:
        owner = payload["repository"]["owner"]["login"]
        repo_name = payload["repository"]["name"]
        full_name = payload["repository"]["full_name"]
        github_repo_id = payload["repository"]["id"]
        pr_number = payload["pull_request"]["number"]
        head_sha = payload["pull_request"]["head"]["sha"]
        installation_id = payload["installation"]["id"]

        # Step 1: Get installation token (per-request, GH-08)
        try:
            private_key = settings.github_private_key.replace("\\n", "\n")
            token = await get_installation_token(settings.github_app_id, private_key, installation_id)
        except Exception as exc:
            logger.error("Token fetch failed for %s PR#%s: %s", full_name, pr_number, exc)
            # Can't post failure comment without a token — just log
            return

        # Step 2: Fetch PR diff
        try:
            diff_text = await fetch_pr_diff(owner, repo_name, pr_number, token)
        except Exception as exc:
            logger.error("Diff fetch failed: %s", exc)
            await post_failure_comment(owner, repo_name, pr_number, str(exc), token)
            return

        # Step 3: Parse diff; build valid comment positions map
        patch = PatchSet(diff_text)
        valid_positions = build_diff_comment_positions(diff_text)

        # Step 4: Upsert Repo record; get repo_id for DB writes
        async with AsyncSessionLocal() as session:
            result = await session.execute(select(Repo).where(Repo.github_repo_id == github_repo_id))
            repo_record = result.scalar_one_or_none()
            if repo_record is None:
                repo_record = Repo(github_repo_id=github_repo_id, repo_name=full_name)
                session.add(repo_record)
                await session.flush()  # assigns repo_record.id
            repo_id = repo_record.id

            # Step 5: Review each changed file
            all_findings = []
            for patched_file in patch:
                if patched_file.is_binary_file or patched_file.is_removed_file:
                    continue
                file_path = patched_file.path
                # Build code snippet from added/context lines in the diff for this file
                lines = [line.value for hunk in patched_file for line in hunk if not line.is_removed]
                code_snippet = "".join(lines)
                if not code_snippet.strip():
                    continue
                # Detect language from file extension
                ext = file_path.rsplit(".", 1)[-1].lower() if "." in file_path else "text"
                lang_map = {"py": "python", "js": "javascript", "ts": "typescript", "go": "go", "java": "java", "rb": "ruby", "rs": "rust"}
                language = lang_map.get(ext, "text")

                try:
                    findings = await run_review(code_snippet, language, settings)
                except ReviewPipelineError as exc:
                    logger.warning("Review failed for %s: %s", file_path, exc)
                    findings = []

                # Save Review record (one per file per PR, DB-03)
                review_record = Review(
                    repo_id=repo_id,
                    pr_number=pr_number,
                    file_path=file_path,
                    code_snippet=code_snippet,
                    findings_json=json.dumps([f.model_dump() for f in findings]),
                )
                session.add(review_record)
                all_findings.extend(findings)

            await session.commit()

        # Step 6: Build inline comments (exclude lines not in diff to prevent 422)
        inline_comments = []
        for patched_file in patch:
            if patched_file.is_binary_file or patched_file.is_removed_file:
                continue
            file_path = patched_file.path
            for finding in all_findings:
                comment = finding_to_comment(finding, file_path, valid_positions)
                if comment:
                    inline_comments.append(comment)

        # Step 7: Format summary and submit review
        summary_body, event = format_summary_comment(all_findings)
        try:
            await submit_review(owner, repo_name, pr_number, head_sha, inline_comments, event, summary_body, token)
        except Exception as exc:
            logger.error("submit_review failed: %s", exc)
            await post_failure_comment(owner, repo_name, pr_number, f"Failed to post review: {exc}", token)
    ```

    IMPORTANT notes:
    - ALL DB writes happen inside a single `async with AsyncSessionLocal() as session:` block — not the request-scoped session (pitfall 4)
    - private_key = settings.github_private_key.replace("\\n", "\n") — handles .env PEM encoding (pitfall 5)
    - findings are associated with files by file_path, but all_findings is a flat list; the inline comment loop must match findings to their file. Simplification: since run_review is called per-file, track (file_path, findings) together as a list of tuples rather than extending all_findings mid-loop. Refactor the Step 5 loop to collect `file_results: list[tuple[str, list[Finding]]]` and flatten for summary at the end.
    - Failure handling: wrap the entire body in try/except Exception as the outermost safety net; log but never let the background task crash silently
